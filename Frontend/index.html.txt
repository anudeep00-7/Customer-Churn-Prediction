    <script>
        document.getElementById('churn-form').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const payload = {
                tenure: parseFloat(document.getElementById('tenure').value),
                monthly_charges: parseFloat(document.getElementById('monthly_charges').value),
                total_charges: parseFloat(document.getElementById('total_charges').value),
                contract_month_to_month: parseInt(document.getElementById('contract').value),
                tech_support_yes: parseInt(document.getElementById('tech_support').value)
            };

            fetch('http://localhost:8002/predict', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            })
            .then(res => res.json())
            .then(data => {
                const resultDiv = document.getElementById('result');
                resultDiv.style.display = 'block';
                
                if (data.churn_prediction === 1) {
                    resultDiv.style.backgroundColor = '#f8d7da';
                    resultDiv.style.color = '#721c24';
                    resultDiv.innerText = `High Risk: ${data.churn_probability}% Probability of Churning`;
                } else {
                    resultDiv.style.backgroundColor = '#d4edda';
                    resultDiv.style.color = '#155724';
                    resultDiv.innerText = `Low Risk: ${data.churn_probability}% Probability of Churning`;
                }
            })
            .catch(err => {
                console.error(err);
            });
        });
    </script>
